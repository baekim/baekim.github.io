def solution(arr1, arr2):
    answer = []
    
    for i in range(len(arr1)):      
        temp = []
        k = 0
        
        for j in range(len(arr1[k])):            
            temp.append(arr1[i][j] + arr2[i][j])
            k += 1
            
        answer.append(temp)
        
    return answer
